### 思路
循环数字
### 代码
```
class Solution {
public:
    int findTheWinner(int n, int k) {
        vector<int> players(n+1);
        int total=n;
        int p=0;
        while(total!=1){
            for(int i=0;i<k;i++){
                p=p%n+1;
                if(players[p]==1){
                    i--;
                }
            }
            players[p]=1;
            total--;
        }
        for(int i=0;i<k;i++){
            p=p%n+1;
            if(players[p]==1){
                i--;
            }
        }
        return p;
    }
};
```

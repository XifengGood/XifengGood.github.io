### 思路
遍历一遍，flag数字计数器加1，遍历flag，大于等于2的加入到结果中。
### 代码
```
class Solution {
public:
    vector<int> findDuplicates(vector<int>& nums) {
        static const int N=1e5+1;
        vector<int> flag(N);
        int n=nums.size();
        for(int i=0;i<n;i++){
            flag[nums[i]]++;
        }
        vector<int> res;
        for(int i=0;i<N;i++){
            if(flag[i]>=2){
                res.push_back(i);
            }
        }
        return res;
    }
};
```

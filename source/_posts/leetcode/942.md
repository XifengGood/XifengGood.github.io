### 思路

### 代码
```
class Solution {
public:
    vector<int> diStringMatch(string s) {
        int top=s.length();
        int bottom=0;
        vector<int> res(top+1);
        for(int i=0;i<s.length();i++){
            if(s[i]=='I'){
                res[i]=bottom++;
            }else if(s[i]=='D'){
                res[i]=top--;
            }
        }
        res[s.length()]=top;
        return res;
    }
};
```

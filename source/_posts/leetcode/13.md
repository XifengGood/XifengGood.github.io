### 思路
switch情况讨论
### 代码
```
class Solution {
public:
    int romanToInt(string s) {
        int n=s.length();
        int res=0;
        for(int i=0;i<n;i++){
            switch(s[i]){
                case 'I':{
                    int next=i+1;
                    if(next<n){
                        char c=s[next];
                        if(c=='V'){
                            res+=4;i++;
                        }else if(c=='X'){
                            res+=9;i++;
                        }else{
                            res++;
                        }
                    }else{
                        res++;
                    }
                    break;
                }
                case 'V':{
                    res+=5;
                    break;
                }
                case 'X':{
                    int next=i+1;
                    if(next<n){
                        char c=s[next];
                        if(c=='L'){
                            res+=40;i++;
                        }else if(c=='C'){
                            res+=90;i++;
                        }else{
                            res+=10;
                        }
                    }else{
                        res+=10;
                    }
                    break;
                }
                case 'L':{
                    res+=50;
                    break;
                }
                case 'C':{
                    int next=i+1;
                    if(next<n){
                        char c=s[next];
                        if(c=='D'){
                            res+=400;i++;
                        }else if(c=='M'){
                            res+=900;i++;
                        }else{
                            res+=100;
                        }
                    }else{
                        res+=100;
                    }
                    break;
                }
                case 'D':{
                    res+=500;
                    break;
                }
                case 'M':{
                    res+=1000;
                    break;
                }
            }
        }
        return res;
    }
};
```
